import { Tabs, TabItem } from '@astrojs/starlight/components';

Installing CommandKit is really simple. Based on your preferred package manager, run one of the commands in your terminal:

<Tabs>
    <TabItem label="npm">
        ```sh
        npm install commandkit
        ```
    </TabItem>
    
    <TabItem label="yarn">
        ```sh
        yarn add commandkit
        ```
    </TabItem>

    <TabItem label="pnpm">
        ```sh
        pnpm add commandkit
        ```
    </TabItem>
</Tabs>
